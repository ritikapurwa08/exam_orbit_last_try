"use client";

import { useEffect, useState } from "react";
import { useRouter } from "next/navigation";
import { auth, db } from "@/lib/firebase";
import { onAuthStateChanged, signOut } from "firebase/auth";
import { doc, getDoc } from "firebase/firestore";

import { Icons } from "@/components/icons";
import { IMAGES } from "@/lib/images";
import { ProfileEditDialog } from "@/components/profile/ProfileEditDialog";
import { Skeleton } from "@/components/ui/skeleton";
import {
  LockIcon,
  MoonIcon,
  ShieldCheck,
  BarChart3,
  History,
  BookOpen,
} from "lucide-react";
import { Switch } from "@/components/ui/switch";

export default function ProfilePage() {
  const router = useRouter();
  // eslint-disable-next-line @typescript-eslint/no-explicit-any
  const [user, setUser] = useState<any>(null);
  const [loading, setLoading] = useState(true);
  const [isAdmin, setIsAdmin] = useState(false);
  const [isEditOpen, setIsEditOpen] = useState(false);

  // Profile Data
  const [displayName, setDisplayName] = useState("");
  const [photoURL, setPhotoURL] = useState("");
  // eslint-disable-next-line @typescript-eslint/no-unused-vars
  const [stats, setStats] = useState({
    joined: "Jan 2026",
    totalTests: 54,
  });

  // eslint-disable-next-line @typescript-eslint/no-explicit-any
  const fetchUserData = async (u: any) => {
    try {
      const userDoc = await getDoc(doc(db, "users", u.uid));
      let userData = { ...u };

      if (userDoc.exists()) {
        const data = userDoc.data();
        userData = { ...userData, ...data };
        if (data?.isAdmin) setIsAdmin(true);
      }

      setUser(userData);
      setDisplayName(userData.displayName || "Student Name");
      setPhotoURL(userData.photoURL || IMAGES.USER_AVATAR);
      return userData;
    } catch (error) {
      console.error("Error fetching user data:", error);
    }
  };

  useEffect(() => {
    const unsub = onAuthStateChanged(auth, async (u) => {
      if (!u) {
        router.push("/login");
        return;
      }

      await fetchUserData(u);
      setLoading(false);
    });
    return () => unsub();
  }, [router]);

  const handleProfileUpdate = async () => {
    if (auth.currentUser) {
      await fetchUserData(auth.currentUser);
    }
  };

  const handleLogout = async () => {
    await signOut(auth);
    router.push("/login");
  };

  if (loading)
    return (
      <div className="min-h-screen bg-black flex items-center justify-center">
        <Skeleton className="h-32 w-32 rounded-full bg-slate-900" />
      </div>
    );

  return (
    <div className="flex flex-col min-h-screen bg-black text-white font-sans pb-24">
      {/* Header */}
      <header className="flex items-center justify-between px-4 py-4 pt-8 bg-black sticky top-0 z-50">
        <div className="flex items-center text-slate-500 font-medium text-sm gap-1">
          <span
            onClick={() => router.push("/")}
            className="cursor-pointer hover:text-white transition-colors"
          >
            Home
          </span>
          <Icons.ChevronRight className="w-3.5 h-3.5" />
          <span className="text-white">Profile</span>
        </div>

        <button
          onClick={() => router.back()}
          className="text-indigo-400 font-bold text-sm tracking-wide active:opacity-70"
        >
          Done
        </button>
      </header>

      <main className="px-5 pt-4 pb-8 space-y-8 max-w-lg mx-auto w-full">
        {/* Profile Section */}
        <section className="flex flex-col items-center">
          <div className="relative mb-4">
            {/* Avatar Ring */}
            <div className="size-32 rounded-full p-1 bg-linear-to-b from-[#bfa268] to-[#1a1a1a] shadow-xl">
              <div
                className="size-full rounded-full bg-cover bg-center border-4 border-black"
                style={{
                  backgroundImage: `url(${photoURL})`,
                  backgroundColor: "#222",
                }}
              />
            </div>

            {/* Edit Button */}
            <button
              onClick={() => setIsEditOpen(true)}
              className="absolute bottom-0 right-1 size-9 rounded-full bg-indigo-600 border-4 border-black flex items-center justify-center text-white shadow-lg active:scale-95 transition-transform"
            >
              <Icons.Edit className="w-4 h-4" />
            </button>
          </div>

          <h1 className="text-2xl font-bold text-white mb-1 tracking-tight">
            {displayName}
          </h1>
          <p className="text-sm text-slate-400 mb-2">{user?.email}</p>

          {user?.bio && (
            <p className="text-sm text-slate-300 max-w-xs text-center mb-3 line-clamp-3">
              {user.bio}
            </p>
          )}

          <span className="text-indigo-400 font-bold text-xs uppercase tracking-widest bg-indigo-500/10 px-3 py-1 rounded-full border border-indigo-500/20">
            Scholar
          </span>
        </section>

        <ProfileEditDialog
          user={user}
          open={isEditOpen}
          onOpenChange={setIsEditOpen}
          onUpdate={handleProfileUpdate}
        />

        {/* Stats Grid */}
        <section className="grid grid-cols-2 gap-3">
          <div className="bg-[#111111] border border-white/5 rounded-2xl p-4 flex items-center gap-4">
            <div className="size-12 rounded-xl bg-indigo-500/10 flex items-center justify-center text-indigo-400">
              <Icons.Calendar className="w-6 h-6" />
            </div>
            <div>
              <p className="text-[10px] uppercase font-bold text-slate-500 tracking-wider mb-0.5">
                Joined
              </p>
              <p className="text-sm font-bold text-white">{stats.joined}</p>
            </div>
          </div>
          <div className="bg-[#111111] border border-white/5 rounded-2xl p-4 flex items-center gap-4">
            <div className="size-12 rounded-xl bg-indigo-500/10 flex items-center justify-center text-indigo-400">
              <Icons.BookOpen className="w-6 h-6" />
            </div>
            <div>
              <p className="text-[10px] uppercase font-bold text-slate-500 tracking-wider mb-0.5">
                Total Tests
              </p>
              <p className="text-sm font-bold text-white">
                {stats.totalTests} Exams
              </p>
            </div>
          </div>
        </section>

        {/* Settings List */}
        <section>
          <h3 className="text-xs font-bold text-slate-500 uppercase tracking-widest mb-4 ml-1">
            Settings
          </h3>
          <div className="bg-[#111111] border border-white/5 rounded-3xl overflow-hidden divide-y divide-white/5">
            {/* Account */}
            <div className="p-4 flex items-center justify-between active:bg-white/5 transition-colors cursor-pointer">
              <div className="flex items-center gap-3">
                <div className="size-10 rounded-full bg-blue-500/20 flex items-center justify-center text-blue-500">
                  <Icons.User className="w-5 h-5" />
                </div>
                <span className="font-bold text-sm text-white">Account</span>
              </div>
              <Icons.ChevronRight className="w-5 h-5 text-slate-600" />
            </div>

            {/* Vocabulary */}
            <div
              onClick={() => router.push("/learn/words")}
              className="p-4 flex items-center justify-between active:bg-white/5 transition-colors cursor-pointer"
            >
              <div className="flex items-center gap-3">
                <div className="size-10 rounded-full bg-pink-500/20 flex items-center justify-center text-pink-500">
                  <BookOpen className="w-5 h-5" />
                </div>
                <span className="font-bold text-sm text-white">Vocabulary</span>
              </div>
              <Icons.ChevronRight className="w-5 h-5 text-slate-600" />
            </div>

            {/* Statistics */}
            <div
              onClick={() => router.push("/statistics")}
              className="p-4 flex items-center justify-between active:bg-white/5 transition-colors cursor-pointer"
            >
              <div className="flex items-center gap-3">
                <div className="size-10 rounded-full bg-violet-500/20 flex items-center justify-center text-violet-500">
                  <BarChart3 className="w-5 h-5" />
                </div>
                <span className="font-bold text-sm text-white">Statistics</span>
              </div>
              <Icons.ChevronRight className="w-5 h-5 text-slate-600" />
            </div>

            {/* History */}
            <div
              onClick={() => router.push("/history")}
              className="p-4 flex items-center justify-between active:bg-white/5 transition-colors cursor-pointer"
            >
              <div className="flex items-center gap-3">
                <div className="size-10 rounded-full bg-orange-500/20 flex items-center justify-center text-orange-500">
                  <History className="w-5 h-5" />
                </div>
                <span className="font-bold text-sm text-white">History</span>
              </div>
              <Icons.ChevronRight className="w-5 h-5 text-slate-600" />
            </div>

            {/* Notifications */}
            <div className="p-4 flex items-center justify-between active:bg-white/5 transition-colors cursor-pointer">
              <div className="flex items-center gap-3">
                <div className="size-10 rounded-full bg-emerald-500/20 flex items-center justify-center text-emerald-500">
                  <Icons.Bell className="w-5 h-5" />
                </div>
                <span className="font-bold text-sm text-white">
                  Notifications
                </span>
              </div>
              <Switch checked={true} />
            </div>

            {/* Dark Mode */}
            <div className="p-4 flex items-center justify-between active:bg-white/5 transition-colors cursor-pointer">
              <div className="flex items-center gap-3">
                <div className="size-10 rounded-full bg-slate-700/50 flex items-center justify-center text-slate-300">
                  <MoonIcon className="w-5 h-5" />
                </div>
                <div>
                  <div className="font-bold text-sm text-white">Dark Mode</div>
                  <div className="text-[10px] font-bold text-indigo-400">
                    Premium OLED Active
                  </div>
                </div>
              </div>
              <LockIcon className="w-4 h-4 text-slate-600" />
            </div>

            {/* Admin Console */}
            {isAdmin && (
              <div
                onClick={() => router.push("/admin/add-question")}
                className="p-4 flex items-center justify-between active:bg-white/5 transition-colors cursor-pointer hover:bg-white/5 border-t border-white/5"
              >
                <div className="flex items-center gap-3">
                  {/* Using red/shield for admin */}
                  <div className="size-10 rounded-full bg-red-500/20 flex items-center justify-center text-red-500">
                    <ShieldCheck className="w-5 h-5" />
                  </div>
                  <span className="font-bold text-sm text-white">
                    Admin Console
                  </span>
                </div>
                <Icons.ChevronRight className="w-5 h-5 text-slate-600" />
              </div>
            )}
          </div>
        </section>

        {/* Log Out */}
        <button
          onClick={handleLogout}
          className="w-full h-14 bg-[#111111] border border-white/5 rounded-2xl flex items-center justify-center gap-2 text-red-500 font-bold hover:bg-red-500/10 transition-colors active:scale-[0.98]"
        >
          <div className="size-6 rounded-md border border-red-500/30 flex items-center justify-center">
            <Icons.LogOut className="w-3.5 h-3.5" />
          </div>
          Log Out
        </button>
      </main>
    </div>
  );
}
